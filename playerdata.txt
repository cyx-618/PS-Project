import random

# Two separate lists
questions = ["2 + 2?", "Capital of France?", "Red Planet?"]
answers = ["4", "Paris", "Mars"]

# 1. Create a list of indices [0, 1, 2]
indices = list(range(len(questions)))

# 2. Shuffle ONLY the indices
random.shuffle(indices)

# 3. Loop through the shuffled indices
for i in indices:
    # Use the shuffled index 'i' to pull the matching pair
    print(f"\nQuestion: {questions[i]}")
    user_input = input("Answer: ")
    
    if user_input.lower() == answers[i].lower():
        print("Correct!")
    else:
        print(f"Wrong! It was {answers[i]}")
